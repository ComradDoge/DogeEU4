namespace = vic_prices


# Iron ships
country_event = {
	id = vic_prices.1
	title = vic_prices.EVTNAME1
	desc = vic_prices.EVTDESC1
	picture = TRADE_GOODS_METALS_AND_MINING_EQUIPENT_eventPicture

	fire_only_once = yes
	
	major = yes

	trigger = {
		capital_scope = { has_port = yes }
		dip_tech = 36
		
	}
	mean_time_to_happen = {
		months = 1
	}
	
	
	option = {
		name = vic_prices.EVTOPTA1
		change_price = {
			trade_goods = naval_supplies
			key = IRON_FLEET
			value = -0.25
			duration = -1
		}
		change_price = {
			trade_goods = iron
			key = IRON_FLEET
			value = 0.2
			duration = -1
		}
		add_years_of_income  = 0.25
		add_prestige = 20
	}
}
