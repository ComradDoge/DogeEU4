		}

			factor = 0.80

		}
		modifier = {
			factor = 0.85
			owner = { stability = 2 }
		}
		modifier = {
			factor = 0.90
			owner = { stability = 1 }
		}
		modifier = {
			factor = 1.20
			NOT = { has_owner_religion = yes }
		}
		modifier = {
			factor = 0.70
			owner = { ruler_has_personality = cruel }
		}
	}
	option = {
		name = roads_events.1.a
		ai_chance = {
			factor = 100
		}
		
		#add_prestige = 10
		#add_legitimacy = 5
		#add_republican_tradition = 5
		#add_horde_unity = 5

			name = "recently_assimilated"
			duration = 3650
		}
		ROOT = {
			add_country_modifier = { 
				name = assimilated_through_road
				duration = 7300
			}
		}
	}
}
