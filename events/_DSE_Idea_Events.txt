namespace = dse_idea_events

country_event = {
	id = dse_idea_events.1
	title = "dse_idea_events_1_1.t"
	desc = "dse_idea_events_1_2.d"
	picture = WESTERNISATION_eventPicture
	hidden = yes
	

	trigger = { 
		has_idea = festung4
		any_owned_province = {
			any_province_in_state = {
				OR = {
					fort_level = 2
                    fort_level = 3
                    fort_level = 4
				}

			}
			NOT = { has_province_modifier = Roman_Castrum }
				
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		every_owned_province = {
			limit = {
				any_province_in_state = {
					OR = {
					fort_level = 2
                    fort_level = 3
                    fort_level = 4
					}
				}
				NOT = { has_province_modifier = Roman_Castrum }
            }
			add_province_modifier = {
				name = Roman_Castrum
				duration = -1
				#hidden = yes
			}
		}
		every_owned_province = {
			limit = {
				NOT = {
					any_province_in_state = {
					OR = {
						fort_level = 2
                    	fort_level = 3
                    	fort_level = 4
					}
					}
				}
				has_province_modifier = Roman_Castrum
            }
			remove_province_modifier = Roman_Castrum
		}
	}
	
	option = {
		name = "dse_idea_events_1_3.a"
		
	}

}
