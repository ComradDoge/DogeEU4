country_event = {
	id = ME_Colombia_Venezuela_Gran_Colombia_Events.1
	title = ME_Colombia_Venezuela_Gran_Colombia_Events.EVTNAME1
	desc = ME_Colombia_Venezuela_Gran_Colombia_Events.EVTDESC1
	picture = DIPLOMACY_eventPicture

	is_triggered_only = yes

	option = {
		name = ME_Colombia_Venezuela_Gran_Colombia_Events1.EVTOPT1

		trigger = {
			NOT = { exists = HAT }
		}

		carribeans_region = {
			limit = {
				owned_by = ROOT
			}
			add_core = HAT
			cede_province = HAT
			remove_core = ROOT
		}
		vassalize = HAT
		HAT = {
			change_primary_culture = ROOT
			change_religion = ROOT
			add_liberty_desire = -100
			hidden_effect = {
				if = {
					limit = {
						ROOT = {
							government = monarchy
						}
					}
					change_government = monarchy
					add_government_reform = feudalism_reform
				}
				else = {
					change_government = republic
					add_government_reform = oligarchy_reform
				}
			}
		}
		custom_tooltip = COL_VNZ_HAT_GAINS_BONUSES_WHILE_VASSAL
		set_country_flag = gran_colombia_haiti_vassal_flag
	}

	option = {
		name = ME_Colombia_Venezuela_Gran_Colombia_Events1.EVTOPT2

		random_owned_province = {
			limit = {
				region = carribeans_region
			}
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
		random_owned_province = {
			limit = {
				region = carribeans_region
			}
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
		random_owned_province = {
			limit = {
				region = carribeans_region
			}
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
		random_owned_province = {
			limit = {
				region = carribeans_region
			}
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
		random_owned_province = {
			limit = {
				region = carribeans_region
			}
			add_base_tax = 1
			add_base_production = 1
			add_base_manpower = 1
		}
	}
}
