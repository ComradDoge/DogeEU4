namespace = ab_croatia

#Slavicization
province_event = {
	id = ab_croatia.1
	title = ab_croatia.1.t
	desc = ab_croatia.1.d
	picture = AB_BYZFALL_eventPicture
	
	hidden = yes
	
	mean_time_to_happen = {
		months = 480
		modifier = {
			factor = 0.5
			owner = { mission_completed = cro_adriatic_empire }
		}
	}
	
	trigger = {
		owner = {
			mission_completed = cro_slavicize_the_italians
		}
		region = italy_region
		NOT = {
			OR = {
				culture = jadranian_culture
				culture = talian_culture
			}
		}
	}
	
	option = {
		name = ab_croatia.1
		if = {
			limit = {
				region = italy_region
				NOT = {
					area = venetia_area
				}
			}
			change_culture = talian_culture
		}
		if = {
			limit = {
				area = venetia_area
			}
			change_culture = jadranian_culture
		}
	}
}
