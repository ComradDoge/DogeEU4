namespace = ai_build_remove

country_event = { # Removes trenches from ai so it doesnt acrue devastation
	id = ai_build_remove.1
	title = ai_build_remove.1.t
	desc = ai_build_remove.1.d
	picture = SCOTTISH_MERCENARIES_eventPicture
	
	hidden = yes
	
	trigger = {
		ai = yes
		is_at_war = no
        any_owned_province = {
            has_building = trenches
        }
	}
	
	immediate = {
       every_owned_province = {
           limit = {
               has_building = trenches
           }
            remove_building = trenches
        }
	}
	
	option = {
		name = "nothing"
	}
}
