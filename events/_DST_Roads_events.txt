### Events for Roads mod ###

#Culture spread from roads
province_event = {
	id = roads_events.1
	title = roads_events.1.title
	desc = roads_events.1.desc
	picture = CITY_VIEW_eventPicture
	goto = ROOT

	trigger = {
		has_owner_culture = no
		is_core = owner
		controlled_by = owner
		has_siege = no
		NOT = { local_autonomy = 0.40 }

		OR = {
			has_building = early_road
			has_building = gravel_road
			has_building = royal_road
			has_building = macadam_road
		}
		OR = {
			any_neighbor_province = {
				owned_by = ROOT
				has_owner_culture = yes
				is_core = owner
			}
			is_capital = yes
		}
	}
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.40
			is_capital = yes
		}
		modifier = {
			factor = 0.70
			owner = { stability = 3 }
		}
		modifier = {
			factor = 0.80
			owner = { stability = 2 }
		}
		modifier = {
			factor = 0.90
			owner = { stability = 1 }
		}
		modifier = {
			factor = 1.20
			NOT = { has_owner_religion = yes }
		}
		modifier = {
			factor = 0.90
			owner = { ruler_has_personality = cruel }
		}
	}
	option = {
		name = roads_events.1.a
		ai_chance = {
			factor = 70 
		}
		add_province_modifier = {
			name = "recently_assimilated"
			duration = 3650
		}

		change_culture = owner
		add_local_autonomy = 10

	}
	option = {
		name = roads_events.1.b
		ai_chance = { factor = 30 }
		add_years_of_income = 3
		add_unrest = 2
		
	}
}

province_event = {
	id = roads_events.2
	title = roads_events.2.title
	desc = roads_events.2.desc
	picture = CITY_DEVELOPMENT_eventPicture

	trigger = {
		OR = { 
			NOT = {
				has_building = early_road
			}
			AND = {
				has_building = early_road
				owner = { adm_tech = 10 }
			}
			AND = {
				has_building = gravel_road
				owner = { adm_tech = 17 }
			}
			AND = { 
				has_building = royal_road
				owner = { adm_tech = 23 }
			}
		}
		NOT = { has_building = railroad }
		NOT = { owner = { has_country_modifier = improved_road } }
		is_core = owner
		controlled_by = owner
		has_siege = no
		NOT = { local_autonomy = 0.3 }
		OR = { 
			base_tax = 7
			development = 14
		}
	}
	mean_time_to_happen = {
		months = 240
		modifier = {
			factor = 0.90
			is_capital = yes
		}
		modifier = {
			factor = 0.85
			owner = { has_idea_group = innovativeness_ideas }
		}
		modifier = {
			factor = 0.85
			owner = { stability = 3}
		}
		modifier = {
			factor = 0.90
			owner = { stability = 2 }
		}
		modifier = {
			factor = 2.5
			NOT = { owner = { stability = 0 } }
		}
		modifier = {
			factor = 1.20
			owner = { stability = 0 }
		}
		modifier = {
			factor = 0.80
			owner = { ruler_has_personality = architectural_visionary_personality }
		}
	}
	option = {
		name = roads_events.2.a
		ai_chance = {
			factor = 70 
		}
		if = {
			limit = {
				AND = {
					has_building = royal_road
					adm_tech = 23
				}
			}
			add_building_construction = {
				building = macadam_road
				speed = 0.8
				cost = 0.2
			}
			#add_building = macadam_road
			remove_building = royal_road
			
		}		
		if = {
			limit = {
				AND = {
					has_building = gravel_road
					adm_tech = 18
				}
			}
			add_building_construction = {
				building = royal_road
				speed = 0.8
				cost = 0.2
			}
			#add_building = royal_road
			remove_building = gravel_road
			

		}
		if = {
			limit = {
				AND = { 
					has_building = early_road 
					adm_tech = 10
				}
			}
			add_building_construction = {
				building = gravel_road
				speed = 0.8
				cost = 0.2
			}
			#add_building = gravel_road
			remove_building = early_road

		}
		if = {
			limit = {
				NOT = {
					has_building = early_road
					has_building = gravel_road
					has_building = royal_road
					has_building = macadam_road
					has_building = railroad
				}
			}
			add_building_construction = {
				building = early_road
				speed = 0.8
				cost = 0.2
			}
			#add_building = early_road
		}

		
		owner = {
			add_country_modifier = {
				name = "improved_road"
				duration = 3650
			}
		}

	}
	option = {
		name = roads_events.2.b
		ai_chance = { factor = 30 }
		add_years_of_income = 2
		add_local_autonomy = 10
	}
}
