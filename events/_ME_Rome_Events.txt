namespace = ME_Rome_Events

country_event = {
	id = ME_Rome_Events.1
	title = ME_Rome_Events.1.t
	desc = ME_Rome_Events.1.d
	picture = RELIGION_eventPicture
	
	hidden = yes
	
	trigger = {
		tag = ROM
		any_owned_province = {
			culture = roman
			has_province_modifier = rome_roman_culture
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					culture = roman
					has_country_modifier = rome_roman_culture
				}
				remove_province_modifier = rome_roman_culture
			}
		}
	}
	
	option = {
		name = "ME_Rome_Events.1.a"
		
		ai_chance = {
			factor = 100
		}
	}
}

country_event = {
	id = ME_Rome_Events.2
	title = ME_Rome_Events.2.t
	desc = ME_Rome_Events.2.d
	picture = RELIGION_eventPicture
	
	hidden = yes
	
	trigger = {
		tag = ROM
		exists = REI
		REI = {
			is_subject_of = ROOT
		}
		any_owned_province = {
			OR = {
				region = deccan_region
				region = hindusthan_region
				region = bengal_region
				region = west_india_region
				region = coromandel_region
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		hidden_effect = {
			every_owned_province = {
				limit = {
					OR = {
						region = deccan_region
						region = hindusthan_region
						region = bengal_region
						region = west_india_region
						region = coromandel_region
					}
				}
				cede_province = REI
			}
		}
	}
	
	option = {
		name = "ME_Rome_Events.2.a"
		
		ai_chance = {
			factor = 100
		}
	}
}
