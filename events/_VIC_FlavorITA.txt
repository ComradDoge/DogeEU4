			limit = {

# PROVINCENAME Revolts!
province_event = {
	id = vic_flavor_ita.5
	title = "vic_flavor_ita.5.t"
	desc = "vic_flavor_ita.5.d"
	picture = ANGRY_MOB_eventPicture
	
	trigger = {
		has_global_flag = young_italy_happened
		owner = { NOT = { tag = ITA } }
		region = italy_region
		culture_group = latin
		OR = {
			NOT = { has_province_flag = italian_revolt }
			had_province_flag = { flag = italian_revolt days = 7300 }
		}
		owner = {
			OR = {
				capital_scope = {
					NOT = { region = italy_region }
				}
				NOT = { culture_group = latin }
				tag = PAP
			}
		}
	}

	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 0.75
			any_neighbor_province = {
				owned_by = ITA
			}
		}
		modifier = {
			factor = 0.75
			any_neighbor_province = {
				has_province_flag = italian_revolt
			}
		}
		modifier = {
			factor = 0.75
			has_owner_religion = no
		}
		modifier = {
			factor = 0.75
			owner = {
				NOT = { culture_group = ROOT }
			}
		}
		modifier = {
			factor = 0.75
			owner = {
				NOT = { primary_culture = ROOT }
				NOT = { accepted_culture = ROOT }
			}
		}
		modifier = {
			factor = 0.5
			owner = { war_with = ITA }
		}
		modifier = {
			factor = 0.75
			unrest = 3
		}
		modifier = {
			factor = 0.75
			unrest = 5
		}
		modifier = {
			factor = 0.75
			unrest = 7
		}
		modifier = {
			factor = 0.75
			unrest = 9
		}
	}

	immediate = {
		hidden_effect = {
			IF = {
				limit = { NOT = { has_global_flag = Garibaldi_Redshirts } }
				set_global_flag = Garibaldi_Redshirts
			}
		}
	}	

	option = {
		name = "vic_flavor_ita.5.a"
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				owner = {
					NOT = { num_of_cities = 5 }
				}
			}
		}
		set_province_flag = italian_revolt
		if = {
			limit = {
				owner = { ai = no }
			}
			spawn_rebels = {
				type = nationalist_rebels
				size = 3
				friend = ITA
			}
		}
		else = {
			spawn_rebels = {
				type = nationalist_rebels
				size = 2
				friend = ITA
			}
		}	
	}
	option = {
		name = "vic_flavor_ita.5.b"
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				owner = {
					num_of_cities = 5
				}
			}
		}
		if = {
			ITA = {
				exists = yes
			}
			cede_province = ITA
		}
		else = {
			add_local_autonomy = 100
		}
	}
}

# Italian Crown

	id = vic_flavor_ita.6
	title = "vic_flavor_ita.6.t"
	desc = "vic_flavor_ita.6.d"
	picture = ANGRY_MOB_eventPicture

	major = yes
	fire_only_once = yes
	
	trigger = {
		has_global_flag = Garibaldi_Redshirts
		ai = yes
		was_player = no
		NOT = { exists = ITA }
		NOT = { tag = PAP }

		is_free_or_tributary_trigger = yes
		num_of_cities = 6
		any_owned_province = {
			region = italy_region
			culture_group = latin
		}
		capital_scope = {
			region = italy_region
		}
		culture_group = latin
		calc_true_if = {
			all_owned_province = {
				region = italy_region
			}
			amount = 6
		}
	}
	
	mean_time_to_happen = {
		months = 360
		modifier = {
			factor = 0.55
			total_development = 150
		}
		modifier = {
			factor = 0.55
			total_development = 200
		}
		modifier = {
			factor = 0.55
			total_development = 250
		}
		modifier = {
			factor = 0.55
			total_development = 300
		}
		modifier = {
			factor = 0.55
			total_development = 350
		}
		modifier = {
			factor = 0.55
			total_development = 400
		}
		modifier = {
			factor = 0.55
			total_development = 450
		}
		modifier = {
			factor = 0.55
			total_development = 500
		}
	}
	
	option = {
		name = "vic_flavor_ita.6.a"
			change_tag = ITA
			if = {
				limit = {
					capital_scope = { is_part_of_hre = yes }
					is_elector = no
					is_emperor = no
				}
				emperor = {
					add_opinion = {
						who = root
						modifier = opinion_left_empire
					}
				}
				set_in_empire = no
				emperor = {
					country_event = { id = hre_event.5 }
				}
			}
			if = {
				limit = {
					NOT = { government_rank = 2 }
				}
				set_government_rank = 2
			}
			if = {
				limit = {
					NOT = { government = monarchy }
				}
				change_government = monarchy
			}
			set_legacy_government = parliamentary_monarchy
			add_prestige = 25
			if = {
				limit = { has_custom_ideas = no }
				country_event = { id = ideagroups.1 } #Swap Ideas
			}
			set_country_flag = formed_italy_flag
		}
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				government = republic
			}
		}
	
	option = {
		name = "vic_flavor_ita.6.b"
		prestige = -20
	}
}
