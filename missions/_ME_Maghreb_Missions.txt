				}
			}
		}
		effect = {

			define_advisor = {
				type = army_reformer

				cost_multiplier = 0.75
			}
			
			custom_tooltip = " "
			
			custom_tooltip = GHD_if_we_have_65_berber_loyalty
			tooltip = {
				add_yearly_manpower = 1.5
			}
			hidden_effect = {
				if = {
					limit = {
						estate_loyalty = {
							estate = estate_berbers
							loyalty = 65
						}
					}
					add_yearly_manpower = 1.5
				}
			}
		}
	}
	
	GHD_army = {

			}
			else = {

			}

						add_fire = 2

			}
