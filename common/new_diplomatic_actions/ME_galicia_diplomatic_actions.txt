gal_send_money = {
	category = influence

	require_acceptance = no

	is_visible = {
		FROM = {
			has_country_flag = me_gal_start_of_revolt
		}
		OR = {
			tag = POR
			tag = CAS
		}
	}

	is_allowed = {
		treasury = 50
		OR = {
			NOT = {
				has_country_flag = ee_gal_support_cooldown
			}
			had_country_flag = {
				flag = ee_gal_support_cooldown
				days = 365
			}
		}
	}

	on_accept = {
		add_treasury = -50
		clr_country_flag = ee_gal_support_cooldown
		set_country_flag = ee_gal_support_cooldown
		FROM = {
			change_variable = {
				which = civil_war_contribution
				value = 10
			}
			if = {
				limit = {
					check_variable = {
						which = civil_war_contribution
						value = 100
					}
				}
				country_event = { id = ME_Galicia_Events.3 }
			}
		}
		if = {
			limit = {
				tag = CAS
			}
			FROM = {
				change_variable = {
					which = cascivil_war_contribution
					value = 10
				}
			}
		}
		else_if = {
			limit = {
				tag = POR
			}
			FROM = {
				change_variable = {
					which = porcivil_war_contribution
					value = 10
				}
			}
		}
	}

	ai_will_do = {
		treasury = 100
		current_income_balance = 1
		OR = {
			tag = CAS
			AND = {
				tag = POR
				OR = {
					NOT = {
						has_country_flag = ee_gal_support_cooldown
					}
					had_country_flag = {
						flag = ee_gal_support_cooldown
						days = 900
					}
				}
			}
		}
	}
}
