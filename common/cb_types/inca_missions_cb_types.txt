cb_inca_expulse_europeans = {
	valid_for_subject = no
	
	prerequisites = {
		OR = {
			mission_completed = inca_expel_the_foreigners
			mission_completed = maya_expel_the_foreigners
		}
		OR = {
			AND = {
				has_dlc = "El Dorado"
				is_religion_reformed = yes
			}
			AND = {
				NOT = { has_dlc = "El Dorado" }
				is_primitive = no
				is_tribal = no
				army_size = 30
			}
			NOT = { religion_group = pagan }
		}
		FROM = {
			OR = {
				capital_scope = {
					continent = europe
				}
				AND = {
					is_colonial_nation = yes
					overlord = {
						capital_scope = {
							continent = europe
						}
					}
				}
			}
		}
		OR = {
			FROM = {
				any_owned_province = {
					OR = {
						continent = north_america
						continent = south_america
					}
					colony_claim = ROOT
				}
			}
			any_owned_province = {
				OR = {
					continent = north_america
					continent = south_america
				}
				OR = {
					any_neighbor_province = {
						owned_by = FROM
					}
					sea_zone = {
						any_neighbor_province = {
							owned_by = FROM
						}
					}
				}
			}
		}
	}

	war_goal = expel_the_foreigners_wg
}
