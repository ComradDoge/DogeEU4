po_EE_Celtica_Change_Religion = {
	category = 6 
	power_projection = no 
	power_cost_base = 0 
	prestige_base = 0.1
	ae_base = 0
	warscore_cost = {
		all_provinces = 0.0
		no_provinces = 25
		owner_keeps = 0.0
		conquered = 0.0
		returned_core = 0.0
		released_nation = 0.0
		cancelled_subject = 0.0
		concede_colonial_region = 0.0
	}
	warscore_cap = 100
	requires_demand_independence = no 
	is_make_subject = no
	requires_is_allowed = yes 
	is_visible = { }
	is_allowed = { }
	effect = { 
		FROM = { change_religion = ROOT }
		if = {
			limit = {
				FROM = { 
					num_of_cities = 1
					NOT = { num_of_cities = 5 }
				}
			}
			FROM = {
				random_owned_province = {
					limit = {
						NOT = { religion = CEE }
					}
					change_religion = CEE
				}
			}
		}
		if = {
			limit = {
				ROOT = { has_country_flag = EE_CLT_new_religious_victory_flag }
			}
			hidden_effect = {
				FROM = { save_event_target_as = EE_CLT_war_loser }
			}
			ROOT = {
				country_event = {
					id = EE_Celtica_Events.10
				}
			}
		}
		else = {
			custom_tooltip = EE_CLT_change_4_religion_root
			hidden_effect = {
				while = {
					limit = {
						FROM = {
							num_of_owned_provinces_with = {
								value = 4
								NOT = { has_province_flag = EE_CLT_change_religion }
							}
						}
					}
					FROM = {
						random_owned_province = {
							limit = {
								NOT = { has_province_flag = EE_CLT_change_religion }
								NOT = { religion = CEE }
							}
							change_religion = CEE
							set_province_flag = EE_CLT_change_religion
						}
					}
				}
			}
		}
	}
	ai_weight = { 
		export_to_variable = {
			variable_name = ai_value 
			value = 10000
		}
	}
}
