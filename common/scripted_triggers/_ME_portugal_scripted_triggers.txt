POR_extra_feitoria_trigger = {
	if = {
		limit = {
			$id$ = { 
				NOT = { has_province_flag = POR_feitoria_is_here }
			}
		}
		$id$ = { 
			owned_by = ROOT
		}
	}
	else = {
		hidden_trigger = {
			always = no
		}
	}
}

POR_extra_feitoria_pth = {
	if = {
		limit = {
			$id$ = {
				NOT = { has_province_flag = POR_feitoria_is_here }
			}
		}
		province_id = $id$
	}
	else = {
		always = no
	}
}

}

POR_preview_trigger = {
	if = {
		limit = {
			has_country_flag = POR_preview
		}
		custom_trigger_tooltip = {
			tooltip = POR_preview_trigger
			always = no
		}
	}
}
