NED_faction_is_superior_by = {
	[[hook]
		has_global_modifier_value = {
			which = dutch_factionalism_balance
			value = $value$
		}
	]
	[[cod]
		NOT = {
			has_global_modifier_value = {
				which = dutch_factionalism_balance
				value = -$value$
			}
		}
	]
}
