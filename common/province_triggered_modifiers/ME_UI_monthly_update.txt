ME_UI_monthly_update = {
	potential = {
		always = yes
	}
	trigger = {
		had_global_flag = {
			flag = ME_ui_monthly
			days = 30
		}
	}
	hidden = yes 
	
	#Province scope
	on_activation = {
		clr_global_flag = ME_ui_monthly
		set_global_flag = ME_ui_monthly
		
		every_country = {
			limit = {
				has_country_flag = ME_reducing_execution_cost
			}
			update_execution_cost_monthly = yes
		}
	}
}
