add_age_modifier = {
	if = {
		limit = {
			NOT = {
				has_global_flag = ME_$age$_ended
			}
		}
		add_country_modifier = {
			name = $name$
			duration = $duration$
			desc = ME_until_the_end_of_$age$
		}
		set_country_flag = ME_got_age_modifier_for_$age$
	}
	else = {
		[[else] $else$ ]
	}
}
