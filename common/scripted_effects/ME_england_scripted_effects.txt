ENG_tally_french_deaths = {
	if = {
		limit = {
			has_country_flag = ENG_hyw_tree_incident
			FROM = {
				check_variable = {
					which = land_units_killed
					value = 1
				}
			}
			FROM = {
				war_with = ROOT
				OR = {
					is_in_war = {
						attacker_leader = FRA
					}
					is_in_war = {
						defender_leader = FRA
					}
				}
			}
		}
		FROM = {
			set_variable = {
				which = eng_needed_casualties
				which = land_units_killed
			}
		}
		change_variable = {
			which = eng_needed_casualties
			which = FROM
		}
	}
}
