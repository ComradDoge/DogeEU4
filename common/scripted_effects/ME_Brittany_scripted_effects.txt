		limit = {
			NOT = {
				is_in_war = { casus_belli = EE_BRI_Mad_war }

			limit = {
				has_country_modifier = EE_BRI_English_Longbow

			clr_country_flag = EE_BRI_ENG_GBR_Mad_War

			}

			}
			remove_country_modifier = EE_BRI_Spanish_Steel

}
