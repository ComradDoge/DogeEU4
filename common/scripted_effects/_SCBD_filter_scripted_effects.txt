SBCD_get_potential_and_excluded_province_effects = {
	set_variable = { which = SBCD_num_of_province_var value = 0 }
	set_variable = { which = SBCD_num_of_excluded_province_var value = 0 }
	set_variable = { which = SBCD_num_of_potential_province_var value = 0 }
	event_target:SBCD_selected_country_target = {
		every_owned_province = {
			ROOT = {
				change_variable = {
					which = SBCD_num_of_province_var
					value = 1
				}
			}
		}
		every_owned_province = {
			limit = { SBCD_potential_province_triggers = yes }
			ROOT = {
				change_variable = {
					which = SBCD_num_of_potential_province_var
					value = 1
				}
			}
		}
		every_owned_province = {
			limit = { 
				OR = {
					SBCD_potential_province_triggers = no
					has_province_flag = SBCD_excluded_flag
				}
			}
			ROOT = {
				change_variable = {
					which = SBCD_num_of_excluded_province_var
					value = 1
				}
			}
		}
	}
}

SBCD_get_province_development_effects = {
	event_target:SBCD_selected_country_target = {
		every_owned_province = {
			limit = { SBCD_potential_province_triggers = yes }
			export_to_variable = {
				which = SBCD_province_$TYPE$_var
				value = trigger_value:$TYPE$
			}
		}
	}
}
SBCD_clear_province_development_var_effects = {
	event_target:SBCD_selected_country_target = {
		every_owned_province = {
			set_variable = { which = SBCD_province_base_tax_var value = 0 }
			set_variable = { which = SBCD_province_base_production_var value = 0 }
			set_variable = { which = SBCD_province_base_manpower_var value = 0 }
			set_variable = { which = SBCD_province_development_var value = 0 }
		}
	}
}
SBCD_exclude_province_effects = {
    event_target:SBCD_selected_country_target = {
        every_owned_province = { clr_province_flag = SBCD_excluded_flag }
    }
    if = {
		limit = { has_country_flag = SBCD_target_state_filter_flag }
		trigger_switch = {
			on_trigger = has_country_flag
			SBCD_target_is_state_flag = {
				event_target:SBCD_selected_country_target = {
					every_owned_province = {
						limit = {
							SBCD_potential_province_triggers = yes
                            is_state = no
						}
						set_province_flag = SBCD_excluded_flag
					}
				}
			}
			SBCD_target_is_trade_company_flag = {
				event_target:SBCD_selected_country_target = {
					every_owned_province = {
						limit = {
							SBCD_potential_province_triggers = yes
                            is_owned_by_trade_company = no
						}
						set_province_flag = SBCD_excluded_flag
					}
				}
			}
			SBCD_target_is_territory_flag = {
				event_target:SBCD_selected_country_target = {
					every_owned_province = {
						limit = {
							SBCD_potential_province_triggers = yes
							OR = {
								is_state = yes
								is_owned_by_trade_company = yes
							}
						}
						set_province_flag = SBCD_excluded_flag
					}
				}
			}
		}
	}

	
    SBCD_exclude_target_development_type_effects = { TYPE = base_tax }
    SBCD_exclude_target_development_type_effects = { TYPE = base_production }
    SBCD_exclude_target_development_type_effects = { TYPE = base_manpower }
	SBCD_exclude_target_development_type_effects = { TYPE = development }

    if = {
        limit = { has_country_flag = SBCD_target_terrain_filter_flag }
        trigger_switch = {
            on_trigger = has_country_flag
            SBCD_target_is_grasslands_flag = { SBCD_exclude_target_terrain_effect = { TERRAIN = grasslands } }
            SBCD_target_is_farmlands_flag = { SBCD_exclude_target_terrain_effect = { TERRAIN = farmlands } }
            SBCD_target_is_drylands_flag = { SBCD_exclude_target_terrain_effect = { TERRAIN = drylands } }

    if = {
		limit = { has_country_flag = SBCD_target_$TYPE$_filter_flag }
		trigger_switch = {
			on_trigger = has_country_flag
			SBCD_target_$TYPE$_equal_to_flag = {
				event_target:SBCD_selected_country_target = {
					every_owned_province = {
						set_variable = { which = SBCD_target_$TYPE$_var which = ROOT }
						if = {
							limit = {
								NOT = {
									is_variable_equal = {
										which = SBCD_target_$TYPE$_var
										which = SBCD_province_$TYPE$_var
									}
								}
							}
							set_province_flag = SBCD_excluded_flag
						}
					}
				}
			}
			SBCD_target_$TYPE$_greater_than_or_equal_to_flag = {
				event_target:SBCD_selected_country_target = {
					every_owned_province = {
						set_variable = { which = SBCD_target_$TYPE$_var which = ROOT }
						if = {
							limit = {
								NOT = {
									check_variable = {
										which = SBCD_province_$TYPE$_var
										which = SBCD_target_$TYPE$_var
									}
								}
							}
							set_province_flag = SBCD_excluded_flag
						}
					}
				}
			}
			SBCD_target_$TYPE$_lesser_than_or_equal_to_flag = {
				event_target:SBCD_selected_country_target = {
					every_owned_province = {
						set_variable = { which = SBCD_target_$TYPE$_var which = ROOT }
						if = {
							limit = {
								NOT = {
									check_variable = {
										which = SBCD_target_$TYPE$_var
										which = SBCD_province_$TYPE$_var
									}
								}
							}
							set_province_flag = SBCD_excluded_flag
						}
					}
				}
			}
		}
		every_owned_province = {
			set_variable = { which = SBCD_target_$TYPE$_var value = 0 }
		}
	}

		every_owned_province = {
			limit = {
				SBCD_potential_province_triggers = yes
				NOT = { has_terrain = $TERRAIN$ }
			}
			set_province_flag = SBCD_excluded_flag
		}
	}
}
