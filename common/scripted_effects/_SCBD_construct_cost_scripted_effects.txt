			event_target:SBCD_every_owned_province_target = { has_building = macadam_road }
		}
		change_variable = {
			which = SBCD_railroad_cost_var
			value = -500
		}
		set_province_flag = SBCD_potential_railroad_from_upgrade_flag
	}
	multiply_variable = {
		which = SBCD_railroad_cost_var
		which = SBCD_effective_build_cost_var
	}
}
