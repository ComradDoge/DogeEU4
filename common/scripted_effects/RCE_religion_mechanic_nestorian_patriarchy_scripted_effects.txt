rce_set_religion_mechanic_nestorian_patriarchy_value = {
	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = nestorian_patriarchy }
		}
		[[value]
			set_variable = { which = rce_religion_mechanic_nestorian_patriarchy_value value = $value$ }
		]
		[[which]
			set_variable = { which = rce_religion_mechanic_nestorian_patriarchy_value which = $which$ }
		]
		hidden_effect = {
			if = {
				limit = {
					rce_has_religion_mechanic_nestorian_patriarchy_value = { value = 200 }
				}
				set_variable = { which = rce_religion_mechanic_nestorian_patriarchy_value value = 200 }
			}
			else_if = {
				limit = {
					NOT = { rce_has_religion_mechanic_nestorian_patriarchy_value = { value = 0 } }
				}
				set_variable = { which = rce_religion_mechanic_nestorian_patriarchy_value value = 0 }
			}
			set_variable = { which = rce_religion_mechanic_nestorian_patriarchy_value_short which = rce_religion_mechanic_nestorian_patriarchy_value }
			round_variable = { which = rce_religion_mechanic_nestorian_patriarchy_value_short value = -1 } # display without decimals
		}
	}
}

rce_export_yearly_religion_mechanic_nestorian_patriarchy_value_gain = {
	hidden_effect = {
		export_to_variable = {
			which = rce_yearly_religion_mechanic_nestorian_patriarchy_value
			value = modifier:rce_yearly_religion_mechanic_nestorian_patriarchy_change
		}
		set_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_modifiers which = rce_yearly_religion_mechanic_nestorian_patriarchy_value }
		# Base:
		change_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value value = 0.5 }
		# Missionaries:
		export_to_variable = {
			which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_missionaries
			value = trigger_value:num_of_missionaries
		}
		multiply_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_missionaries value = 0.5 }
		change_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_missionaries }
		# Tolerance of True Faith:
		export_to_variable = {
			which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_ttf
			value = modifier:tolerance_own
		}
		change_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_ttf value = -4 }
		multiply_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_ttf value = 0.5 }
		change_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_ttf }
		# Stability:
		if = {
			limit = {
				stability = 1
			}
			set_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_stability value = 1 }
			change_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_stability }
		}
		else_if = {
			limit = {
				NOT = { stability = 0 }
			}
			set_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_stability value = -1 }
			change_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_stability }
		}
		else = {
			set_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_from_stability value = 0 }
		}

		# Opinion of Constantinople Patriarchy
		if = {
			limit = {
				exists = NPA
				NOT = { tag = NPA }
				check_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value value = 0 }
			}
			export_to_variable = {
				which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_modifier_from_opinion
				value = opinion
				who = NPA
				with = ROOT
			}
			divide_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_modifier_from_opinion value = 2 }
			divide_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_modifier_from_opinion value = 100 }
			change_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_modifier_from_opinion value = 1 }
		}
		else = {
			set_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_modifier_from_opinion value = 1 }
		}
		multiply_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_modifier_from_opinion }
		change_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_modifier_from_opinion value = -1 } # display
		multiply_variable = { which = rce_yearly_religion_mechanic_nestorian_patriarchy_value_modifier_from_opinion value = 100 } # display
	}
}

	if = {
		limit = {
			rce_uses_religion_mechanic = { mechanic = nestorian_patriarchy }
		}
		custom_tooltip = rce_set_religion_mechanic_nestorian_patriarchy_patriarchy_controller_tt
		hidden_effect = {
			every_country_including_inactive = {
				clr_country_flag = rce_religion_mechanic_nestorian_patriarchy_is_patriarchy_controller
				remove_country_modifier = rce_religion_mechanic_nestorian_patriarchy_patriarch_controller_modifier
				clr_country_flag = rce_religion_mechanic_nestorian_patriarchy_main_center_of_reformation_selection_window_open
				if = {
					limit = {
						NOT = { tag = PREV }
					}
					remove_country_modifier = rce_religion_mechanic_nestorian_patriarchy_main_interaction_3_modifier
					remove_country_modifier = rce_religion_mechanic_nestorian_patriarchy_main_interaction_6_modifier
				}
			}
			save_global_event_target_as = rce_religion_mechanic_nestorian_patriarchy_patriarch_controller
			set_country_flag = rce_religion_mechanic_nestorian_patriarchy_is_patriarchy_controller
			add_country_modifier = {
				name = rce_religion_mechanic_nestorian_patriarchy_patriarch_controller_modifier
				duration = -1
			}
			if = {
				limit = {
					NOT = { exists = NPA }
				}
				set_global_flag = rce_religion_mechanic_nestorian_patriarchy_patriarch_ruler_name_saved
				NPA = {
					set_saved_name = {
						key = rce_religion_mechanic_nestorian_patriarchy_patriarch_ruler_name
						type = advisor
					}
				}
			}
			else_if = {
				limit = {
					has_global_flag = rce_religion_mechanic_nestorian_patriarchy_patriarch_ruler_name_saved
				}
				clr_global_flag = rce_religion_mechanic_nestorian_patriarchy_patriarch_ruler_name_saved
				NPA = {
					clear_saved_name = rce_religion_mechanic_nestorian_patriarchy_patriarch_ruler_name
				}
			}
			NPA = {
				export_to_variable = {
					which = rce_religion_mechanic_nestorian_patriarchy_patriarch_ruler_year
					value = trigger_value:is_year
				}
			}
			country_event = { id = rce_religion_mechanic_nestorian_patriarchy_events.1 }
			rce_religion_mechanic_nestorian_patriarchy_clear_center_of_reformation = yes
			play_sound = "papal_controller"
			every_country = {
				limit = {
					ai = no
					rce_uses_religion_mechanic = { mechanic = nestorian_patriarchy }
					NOT = { rce_religion_mechanic_nestorian_patriarchy_is_patriarchy_controller = yes }
				}
				play_sound = "new_pope_selected"
			}

			if = {
				limit = {
					has_ruler_modifier = rce_religion_mechanic_nestorian_patriarchy_excommunicated_modifier
				}
				remove_country_modifier = rce_religion_mechanic_nestorian_patriarchy_excommunicated_modifier
			}
			clr_global_flag = rce_religion_mechanic_nestorian_patriarchy_no_new_center_of_reformation
		}
	}
}
