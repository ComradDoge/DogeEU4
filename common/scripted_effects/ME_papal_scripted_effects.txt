    if = {
		limit = {
			BYZ = {
				is_subject_of = ROOT
			}
		}
		BYZ = {
			change_tag = LAE
		}
		LAE = { 
			change_religion = catholic
			set_dynasty = ROOT
			set_country_flag = is_LAE_flag
			hidden_effect = { 
				every_owned_province = { 
					limit = {
						region = balkan_region
						has_owner_religion = no 
					} 
					change_religion = catholic 
					change_culture = ROOT
				}
			}
		}
	}
	else = {
		thrace_area = {
			add_core = LAE 
		}
		release = LAE 
		vassalize = LAE
		LAE = { 
			change_religion = catholic
			set_dynasty = ROOT
			set_country_flag = is_LAE_flag
			hidden_effect = { 
				every_owned_province = { 
					limit = {
						region = balkan_region
						has_owner_religion = no 
					} 
					change_religion = catholic 
					change_culture = ROOT
				}
			}
		}
	}
}
