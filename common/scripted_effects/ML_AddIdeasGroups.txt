ML_on_country_vassal_released = {
	log = test.4
	FROM = {
		# save the number of ideas in variables
		GetAllIdeas = yes
		PREV = {
		set_variable = {
			which = numOfFreeIdeas
			which = PREV
		} }
		country_event = {
			id = ML_FreeIdeasForAll.1
			days = 50
		}
	}
}
ML_give_free_idea_effect = {
	log = displaynumofideas.t
	if = { 	limit = { check_variable = {
		which = numOfFreeIdeas
		value = 1
	}}
		log = test.5
		#add_idea_group = administrative_ideas
		if = { #loop
			limit = { check_variable = {
				which = numOfFreeIdeas
				value = 1
				} }
				ideaorderSmart1 = yes
		}
		if = {
			limit = {
				check_variable = {
					which = numOfFreeIdeas
					value = 1
					}
			}
			log = ErrorText.4
		}
	}
}

export_num_ideas ={
	if = {
		limit = {
			$which$ = 7
		}
		change_variable = {
			which = $variable_name$
			value = 7
		}
	} else_if = {
		limit = {
			$which$ = 6
		}
		change_variable = {
			which = $variable_name$
			value = 6
		}
	}	else_if = {
		limit = {
			$which$ = 5
		}
		change_variable = {
			which = $variable_name$
			value = 5
		}
	}	else_if = {
		limit = {
			$which$ = 4
		}
		change_variable = {
			which = $variable_name$
			value = 4
		}

	}	else_if = {
		limit = {
			$which$ = 3
		}
		change_variable = {
			which = $variable_name$
			value = 3
		}

	} else_if = {
		limit = {
			$which$ = 2
		}
		change_variable = {
			which = $variable_name$
			value = 2
		}

	} else_if = {
		limit = {
			$which$ = 1
		}
		change_variable = {
			which = $variable_name$
			value = 1
		}

	}

}
