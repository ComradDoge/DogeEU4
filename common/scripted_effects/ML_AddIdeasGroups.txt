export_num_ideas ={
	if = {
		limit = {
			$which$ = 7
		}
		change_variable = {
			which = $variable_name$
			value = 7
		}
	} else_if = {
		limit = {
			$which$ = 6
		}
		change_variable = {
			which = $variable_name$
			value = 6
		}
	}	else_if = {
		limit = {
			$which$ = 5
		}
		change_variable = {
			which = $variable_name$
			value = 5
		}
	}	else_if = {
		limit = {
			$which$ = 4
		}
		change_variable = {
			which = $variable_name$
			value = 4
		}

	}	else_if = {
		limit = {
			$which$ = 3
		}
		change_variable = {
			which = $variable_name$
			value = 3
		}

	} else_if = {
		limit = {
			$which$ = 2
		}
		change_variable = {
			which = $variable_name$
			value = 2
		}

	} else_if = {
		limit = {
			$which$ = 1
		}
		change_variable = {
			which = $variable_name$
			value = 1
		}

	}

}
