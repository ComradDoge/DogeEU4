xorme_ai_build_production_buildings_effect = {
	[[production]
        if = {
            limit = {
                can_build = counting_house
            }
            add_building_construction = {
                building = counting_house
                speed = 1
                cost = 1
            }
        }
        else_if = {
            limit = {
                can_build = workshop
            }
            add_building_construction = {
                building = workshop
                speed = 1
                cost = 1
            }
        }
        else = {}
	]
}
