timurid_prince = {
	icon = "muslim"
	potential = {
		dynasty = "Timurid"
		NOT = { tag = TIM }
		NOT = { tag = MUG }
		NOT = { vassal_of = TIM }
	}
	trigger = {
		dynasty = "Timurid"
		NOT = { tag = TIM }
		NOT = { tag = MUG }
		NOT = { vassal_of = TIM }
	}
	valid_for_nation_designer = yes
	nation_designer_cost = 50
	fixed_rank = 2
	maintain_dynasty = yes
	
	modifiers = {
		land_forcelimit_modifier = 0.2
	}
	
	custom_attributes  = {
		brotherly_rivalry = yes
	}
	
	effect = {
		hidden_effect = {
			every_known_country = {
				limit = {
					dynasty = "Timurid"
					NOT = {
						is_subject_of = ROOT
					}
				}
				add_opinion = {
					who = ROOT
					modifier = TIM_sibling_rivalry 
				}
				reverse_add_opinion = {
					who = ROOT
					modifier = TIM_sibling_rivalry 
				}
			}
		}
	}
	
	removed_effect = {
		hidden_effect = {
			every_known_country = {
				limit = {
					dynasty = "Timurid"
				}
				remove_opinion = {
					who = ROOT
					modifier = TIM_sibling_rivalry
				}
				reverse_remove_opinion = {
					who = ROOT
					modifier = TIM_sibling_rivalry 
				}
			}
		}
	}
}
