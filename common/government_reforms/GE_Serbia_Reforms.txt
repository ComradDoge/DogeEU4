serbian_empire_reform = {
    icon = serbian_empire
    allow_normal_conversion = yes
    valid_for_nation_designer = yes
    nation_designer_cost = 0
	potential = {
		OR = {
			has_reform = serbian_empire_reform
			have_had_reform = serbian_empire_reform
		}
	}
    trigger = {
		if = {
			limit = { has_updated_culture = { culture = slavic } }
			primary_culture = serbian_new
		}
		else = {
			primary_culture = serbian
		}
    }
	effect = {
		hidden_effect = {
			every_owned_province = {
				ge_provinceupdate_royal_mining_monopoly = yes
			}
		}
	}
	removed_effect = {
		hidden_effect = {
			every_owned_province = {
				remove_province_modifier = GE_Mining_Monopoly
			}
		}
	}
    modifiers = {
		max_absolutism = 10
		unjustified_demands = -0.25
    }
	custom_attributes = {
		enables_royal_mining_monopoly = yes
	}
    ai = {
        factor = 101
    }
}
