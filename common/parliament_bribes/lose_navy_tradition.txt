lose_navy_tradition = {
	trigger = {
		FROM = { has_port = yes }
		navy_tradition = 5
	}
	
	effect = {
		if = {
			limit = { halved_parliament_bribes = yes }
			add_navy_tradition = -1.5
		}
		else = {
			add_navy_tradition = -3
		}
	}
	
	ai_will_do = {
		factor = 1
		modifier = {
			factor = 0.1
			is_at_war = yes
		}
	}
}
