#Discover X in America
estate_settlers_discover_X_in_america = {
	can_select = {
		OR = {
			num_of_conquistadors = 1
			num_of_explorers = 1
		}
		num_of_ports = 1
		any_province = {
			range = root
			OR = {
				continent = north_america
				continent = south_america
				continent = new_world
			}
			has_port = yes
			is_wasteland = no
			is_city = no
			NOT = { has_discovered = root }
			area_for_scope_province = {
				type = all
				NOT = { has_discovered = root }
			}
		}
	}
	selection_weight = {
		factor = 2.5
		modifier = {
			factor = 0.5
			NOT = { num_of_colonists = 1 }
		}
		modifier = {
			factor = 2
			num_of_colonists = 2
		}
		modifier = {
			factor = 2
			OR = {
				num_of_conquistadors = 2
				num_of_explorers = 2
			}
		}
		modifier = {
			factor = 0.2
			has_country_modifier = colonial_enthusiasm
		}
	}
	pre_effect = {
		random_province = {
			limit = {
				range = root
				OR = {
					continent = north_america
					continent = south_america
					continent = new_world
				}
				has_port = yes
				is_wasteland = no
				is_city = no
				NOT = { has_discovered = root }
				area_for_scope_province = {
					type = all
					NOT = { has_discovered = root }
				}
			}
			save_event_target_as = agenda_province
		}
	}
	task_requirements = {
		event_target:agenda_province = {
			area_for_scope_province = {
				custom_trigger_tooltip = {
					tooltip = discovered_by_root
					has_discovered = root
				}
			}
		}
	}
	task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_settlers
		}
		add_estate_loyalty = {
			estate = estate_settlers
			loyalty = 10
		}
		
		if = {
			limit = {
				has_country_modifier = colonial_enthusiasm
			}
			add_dip_power = 50
		}
		else = {
			add_country_modifier = {
				name = colonial_enthusiasm
				duration = 3650
			}
		}
	}
	fail_if = {	
		NOT = { num_of_explorers = 1 }
		OR = {
			NOT = { num_of_ports = 1 }
			has_not_any_exploration_ideas = yes
		}
	}
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_settlers
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

estate_settlers_hunt_for_seven_cities = {
	can_select = {
		has_dlc = "El Dorado"
		OR = {
			num_of_conquistadors = 1
			num_of_explorers = 1
		}
		any_province = {
			OR = {
				continent = north_america
				continent = south_america
			}
			has_discovered = ROOT
		}
		num_of_ports = 1
		calc_true_if = {
			all_province = {
				NOT = { has_discovered = ROOT }
				is_empty = yes
				is_wasteland = no
				OR = {
					continent = north_america
					continent = south_america
				}
			}
			amount = 30
		}
		NOT = { has_country_modifier = estate_settlers_the_hunt }
	}
	selection_weight = {
		factor = 2.5
		modifier = {
			factor = 2
			num_of_conquistadors = 2
		}
	}
	pre_effect = {
		set_variable = {
			which = estate_settlers_provinces_to_explore
			value = 15
		}
	}
	provinces_to_highlight = {
		OR = {
			continent = north_america
			continent = south_america
		}
		NOT = { has_discovered = ROOT }
	}
	task_requirements = {
		custom_trigger_tooltip = {
			tooltip = estate_settlers_provinces_to_explore_tt
			NOT = {
				check_variable = {
					which = estate_settlers_provinces_to_explore
					value = 1
				}
			}
		}
	}
	task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_settlers
		}
		add_estate_loyalty = {
			estate = estate_settlers
			loyalty = 10
		}

			name = estate_settlers_the_hunt
			duration = 3650
		}
	}
	fail_if = {
		OR = {
			NOT = { num_of_ports = 1 }
			NOT = { can_explore_and_search_for_seven_cities = yes }
		}
	}
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_settlers
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}
