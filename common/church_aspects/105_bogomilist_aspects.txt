
bogomilist_dissolve_monasteries_aspect = {
	sprite = "GFX_dissolve_monasteries_icon"
	cost = 100
	trigger = {
        any_owned_province = {
            OR = { 
                has_building = temple
                has_building = cathedral
            }
		}
	}
	effect = {
        add_years_of_income = 0.25
		custom_tooltip = bogomilist_dissolve_monasteries_aspect_tt
		hidden_effect = {
            random_owned_province = {
                limit = {
                    OR = {
                        has_building = temple
                        has_building = cathedral
                    }
                }
                remove_building = temple
                remove_building = cathedral
            }
		}
	}
	ai_will_do = {
		factor = 2
		modifier = {
			factor = 0
			treasury = 10000
		}
		modifier = {
			factor = 2
			is_in_large_debt = yes
		}
	}
}

		}
	}
}

bogomilist_abstaining_aspect = {
	sprite = "GFX_innovativeness_aspect"
	cost = 50
	trigger = {
        any_owned_province = {
			NOT = { has_province_flag = has_latent_good }
			OR = {
				trade_goods = champagne
				trade_goods = wine
				trade_goods = exotic_animals
				trade_goods = pearls
				trade_goods = rum
				trade_goods = beer
				trade_goods = opium
				trade_goods = incense
			}
		}
	}
	effect = {
		custom_tooltip = bogomilist_abstaining_aspect_tt
		hidden_effect = {
            random_owned_province = {
                limit = {
					NOT = { has_province_flag = has_latent_good }
					OR = {
						trade_goods = champagne
						trade_goods = wine
						trade_goods = exotic_animals
						trade_goods = pearls
						trade_goods = rum
						trade_goods = beer
						trade_goods = opium
						trade_goods = incense
					}
                }
				change_trade_goods = unknown
            }
		}
	}
	ai_will_do = {
		factor = 1
	}
}
