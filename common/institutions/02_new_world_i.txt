		OR = {
			continent = europe
			continent = asia
			continent = africa
		}
		has_port = yes
		is_island = no
		is_state = yes
		OR = {
			province_has_center_of_trade_of_level = 1
			development = 12
			is_capital = yes
		}
		is_in_capital_area = yes
		owner = { 
			has_any_exploration_idea = yes

			OR = {
				any_owned_province = {
					owner = {
						capital_scope = {
							continent = europe
						}
					}
					OR = {
						continent = north_america
						continent = south_america
						continent = new_world
					}
				}

					OR = {
						is_subject_of_type = colony

						OR = {
							continent = north_america
							continent = south_america
							continent = new_world
						}
					}

			factor = 0.5
			scale = yes

				}
			}

				tooltip = tooltip_colonial_overlord_new_world_i
				OR = {
					has_port = yes
					is_capital = yes

			factor = 2

						technology_group = sub_saharan
