march_revolution = {
	potential = {
		always = yes
		NOT = { has_country_flag = had_revolution }
		NOT = { has_global_flag = march_revolution }
		revolution_target_exists = yes
		NOT = { exists = GRP }
		NOT = { tag = PAP }
		NOT = { tag = HAB }
		NOT = { government = prussian_monarchy }
		NOT = { government = celestial_empire }
		has_institution = nationalism_institution
		is_free_or_tributary_trigger = yes
		uses_doom = no
		capital_scope = { continent = europe }
		any_owned_province = {
			OR = {
				culture = bavarian
				culture = franconian
				culture = pommeranian
				culture = prussian
				culture = hessian
				culture = saxon
				culture = swabian
				culture = hannoverian
			}
		}	
	}
	
	can_start = {
		current_age = industrial_age
		any_owned_province = {
			development = 20
			is_capital = no	
			OR = {
				culture = bavarian
				culture = franconian
				culture = pommeranian
				culture = prussian
				culture = hessian
				culture = saxon
				culture = swabian
				culture = hannoverian
			}
		}
		has_any_disaster = no
		OR = {
			NOT = { stability =	0 }
			num_of_loans = 50
			inflation = 10
			corruption = 10
			war_exhaustion = 10
			NOT = { prestige = 0 }
		}
	}
	
	can_stop = { 
		OR = {
			has_any_disaster = yes
			stability = 3
		}
	}
	
	
	progress = {
		modifier = {
			factor = 1
			NOT = { prestige = 0 }
		}
		modifier = {
			factor = 1
			war_exhaustion = 5
		} 
		modifier = {
			factor = 0.5
			num_of_loans = 10
			NOT = { num_of_loans = 20 }
		}	
		modifier = {
			factor = 1
			absolutism = 65
		}
		modifier = {
			factor = 2
			absolutism = 75
		}
		modifier = {
			factor = 3
			absolutism = 85
		}
		modifier = {
			factor = 1
			num_of_loans = 20
		}
		modifier = {
			factor = 2
			is_bankrupt = yes
		}		

	}
	
	can_end = {
		stability = 2
		NOT = { num_of_rebel_controlled_provinces = 1 }
	}
	

	modifier = {
		global_unrest = 10
		stability_cost_modifier = 0.1
	}	
	
	on_start = march_revolution.1
	on_end = march_revolution.9
	
	on_monthly = {
		events = {
			march_revolution.10
		}
		random_events = { 
			1000 = 0
		#	100 = revolution.2
		#	100 = revolution.3
		#	100 = revolution.4
		#	100 = revolution.5
		#	100 = revolution.6
		#	100 = revolution.7
		#	100 = revolution.8
		
		}
	}
}
