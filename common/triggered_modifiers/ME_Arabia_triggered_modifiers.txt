ARB_china_trade = {
	potential = {
		has_country_flag = ARB_china_trade_flag
	}
	trigger = {
		if = {
			limit = {
				exists = event_target:EmperorOfChina
			}
			OR = {
				is_emperor_of_china = yes
				event_target:EmperorOfChina = {
					has_opinion = {
						who = ROOT
						value = 75
					}
				}
			}
		}
		else = {
			1816 = {
				OR = {
					country_or_non_sovereign_subject_holds = ROOT

						has_opinion = {
							who = ROOT
							value = 75
						}
					}
				}
			}
		}
	}
	
	trade_steering = 0.1
	global_prosperity_growth = 0.25
}
